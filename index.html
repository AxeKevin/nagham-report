SQL3 SQL3 · 广告消耗（4渠道 UNION）
📋 表：ks_origin_nova_sync_db.s_hib_{tiktok/facebook/google/snapchat}_advertise_daily_report 📅 范围：近 14 天（全量表，dt 只取最新分区）

合并 TikTok / Facebook / Google / Snapchat 四渠道的每日广告消耗（USD）。

  • 全量表（snapshot），dt 只取最新分区，date_num 才是数据日期
  • TikTok date_num 格式 YYYY-MM-DD（string）
  • Facebook/Google/Snapchat date_num 格式 yyyyMMdd（bigint）
  • Snapchat spend 单位 micro-cents，÷1e6 转 USD
SELECT af_media_source_mapped, p_date, SUM(spend_usd) AS spend_usd FROM (
  -- TikTok: spend 字段单位 USD（string），date_num 格式 YYYY-MM-DD
  SELECT 'tiktokglobal_int' AS af_media_source_mapped,
         date_num AS p_date,
         SUM(CAST(spend AS DOUBLE)) AS spend_usd
  FROM ks_origin_nova_sync_db.s_hib_tiktok_advertise_daily_report
  WHERE dt = '{T1_DASH}'
    AND date_num >= '{START_14D_DASH}' AND date_num <= '{T1_DASH}'
  GROUP BY date_num
  UNION ALL
  -- Facebook: spend USD (string), date_num bigint yyyyMMdd
  SELECT 'Facebook Ads', CAST(date_num AS STRING),
         SUM(CAST(spend AS DOUBLE))
  FROM ks_origin_nova_sync_db.s_hib_facebook_advertise_daily_report
  WHERE dt = '{T1_DASH}'
    AND date_num >= {START_14D} AND date_num <= {T1}
  GROUP BY date_num
  UNION ALL
  -- Google: cost_micros (string, ÷1e6 = USD), date_num bigint yyyyMMdd
  SELECT 'googleadwords_int', CAST(date_num AS STRING),
         SUM(CAST(cost_micros AS DOUBLE) / 1000000.0)
  FROM ks_origin_nova_sync_db.s_hib_google_advertise_daily_report
  WHERE dt = '{T1_DASH}'
    AND date_num >= {START_14D} AND date_num <= {T1}
  GROUP BY date_num
  UNION ALL
  -- Snapchat: spend bigint micro-cents (÷1e6 = USD), date_num bigint yyyyMMdd
  SELECT 'snapchat_int', CAST(date_num AS STRING),
         SUM(spend / 1000000.0)
  FROM ks_origin_nova_sync_db.s_hib_snapchat_creative_daily_report
  WHERE dt = '{T1_DASH}'
    AND date_num >= {START_14D} AND date_num <= {T1}
  GROUP BY date_num
) t
GROUP BY af_media_source_mapped, p_date
ORDER BY p_date DESC, spend_usd DESC
